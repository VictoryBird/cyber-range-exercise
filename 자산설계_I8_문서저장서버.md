## 1. 개요

### 1.1 자산 목적

문서 저장 서버는 군 내부 공유 문서를 저장/관리하는 자체 개발 웹 애플리케이션이다. 군 직원은 업무 문서를 업로드, 검색, 다운로드할 수 있으며, 관리자는 사용자 및 파일 권한을 관리한다.

### 1.2 훈련에서의 역할 (STEP 4-5)

이 자산은 레드팀의 **Path Traversal 및 인증 우회 공격 대상**으로 설계되었다. 자체 개발 웹 애플리케이션에 의도적으로 삽입된 취약점을 통해 시스템 파일 접근, 설정 파일 탈취, 군 내부 문서 대량 다운로드가 가능하다.

### 1.3 핵심 공격 벡터 요약

| 취약점 | 엔드포인트 | 공격 효과 |
|--------|-----------|-----------|
| Path Traversal | GET /api/download?file=../../etc/passwd | 시스템 파일 접근 |
| 무인증 파일 목록 | GET /api/files/list | 전체 문서 목록 노출 |
| 무인증 관리자 API | GET /api/admin/users | 전체 사용자 정보 노출 |
| JWT 검증 미흡 | Authorization 헤더 | JWT 없이도 API 접근 가능 |

### 6.6 auth.py (취약한 인증 모듈)

```python
"""
인증 모듈
⚠ 의도적 취약점: JWT 검증이 선택적(Optional)으로 구현됨
"""

import jwt
from datetime import datetime, timedelta
from typing import Optional
from fastapi import Depends, HTTPException, Header
from passlib.context import CryptContext
from sqlalchemy.orm import Session

from config import settings
from database import get_db
from models import User

pwd_context = CryptContext(schemes=["bcrypt"], deprecated="auto")


def hash_password(password: str) -> str:
    return pwd_context.hash(password)


def verify_password(plain: str, hashed: str) -> bool:
    return pwd_context.verify(plain, hashed)


def create_access_token(data: dict) -> str:
    to_encode = data.copy()
    expire = datetime.utcnow() + timedelta(seconds=settings.JWT_EXPIRATION)
    to_encode.update({"exp": expire})
    return jwt.encode(to_encode, settings.JWT_SECRET, algorithm=settings.JWT_ALGORITHM)


def get_current_user_optional(
    authorization: Optional[str] = Header(None),
    db: Session = Depends(get_db)
) -> Optional[User]:
    """
    ⚠ 취약: JWT가 없으면 None을 반환하지만, API에서 None 검사를 하지 않음
    정상적으로는 인증 실패 시 401을 반환해야 함
    """
    if not authorization:
        return None  # ⚠ 인증 없이 통과

    try:
        scheme, token = authorization.split(" ", 1)
        if scheme.lower() != "bearer":
            return None

        payload = jwt.decode(token, settings.JWT_SECRET, algorithms=[settings.JWT_ALGORITHM])
        username = payload.get("sub")
        if not username:
            return None

        user = db.query(User).filter(User.username == username).first()
        return user
    except (jwt.PyJWTError, ValueError):
        return None  # ⚠ 토큰 오류 시에도 None 반환 (401이 아님)


def get_current_user_required(
    authorization: Optional[str] = Header(None),
    db: Session = Depends(get_db)
) -> User:
    """인증 필수 — 이 함수를 사용해야 하지만 일부 엔드포인트에서 누락됨"""
    user = get_current_user_optional(authorization, db)
    if not user:
        raise HTTPException(status_code=401, detail="인증이 필요합니다")
    return user
```

    user: dict


@router.post("/login", response_model=LoginResponse)
def login(req: LoginRequest, db: Session = Depends(get_db)):
    user = db.query(User).filter(User.username == req.username).first()

    if not user or not verify_password(req.password, user.password_hash):
        # 감사 로그 기록
        db.add(AuditLog(
            action="login_failed",
            username=req.username,
            ip_address="(from request)",
            details=f"로그인 실패: {req.username}"
        ))
        db.commit()
        raise HTTPException(status_code=401, detail="사용자명 또는 비밀번호가 올바르지 않습니다")

    # 마지막 로그인 시간 업데이트
    user.last_login = datetime.utcnow()
    db.add(AuditLog(
        action="login",
        username=user.username,
        details="로그인 성공"
    ))
    db.commit()

    token = create_access_token({"sub": user.username, "role": user.role})

    return LoginResponse(
        access_token=token,
        user={
            "id": user.id,
            "username": user.username,
            "name": user.name,
            "department": user.department,
            "role": user.role,
        }
    )

⚠ 의도적 취약점:
  1. /api/files/list — 인증 없이 접근 가능
  2. /api/download — Path Traversal (../../ 경로 미검증)
  3. JWT 검증 선택적

from typing import Optional

@router.get("/download")
def download_file(
    file: str = Query(..., description="다운로드할 파일명"),
    request: Request = None,
    user: Optional[User] = Depends(get_current_user_optional),
    db: Session = Depends(get_db),
):
    """
    파일 다운로드
    ⚠ 취약: Path Traversal — 파일명에 ../../ 포함 시 시스템 파일 접근 가능
    ⚠ 취약: JWT 검증이 Optional이므로 인증 없이도 다운로드 가능
    """
    # ⚠ 취약 코드: os.path.join은 상대 경로를 해석하지만,
    # 사용자 입력의 ../ 를 정규화하여 상위 디렉토리 접근 허용
    file_path = os.path.join(settings.UPLOAD_DIR, file)

    # ⚠ 취약: os.path.realpath() 또는 경로 검증 없음
    # 정상 코드라면 다음과 같이 검증해야 함:
    #   real_path = os.path.realpath(file_path)
    #   if not real_path.startswith(os.path.realpath(settings.UPLOAD_DIR)):
    #       raise HTTPException(status_code=403, detail="접근 거부")

    if not os.path.exists(file_path):
        raise HTTPException(status_code=404, detail="파일을 찾을 수 없습니다")

    if not os.path.isfile(file_path):
        raise HTTPException(status_code=400, detail="유효한 파일이 아닙니다")

    # 감사 로그 (사용자 정보 없이도 기록)
    db.add(AuditLog(
        action="download",
        username=user.username if user else "anonymous",
        target=file,
        ip_address=request.client.host if request else "unknown",
        details=f"파일 다운로드: {file}"
    ))
    db.commit()

    return FileResponse(
        path=file_path,
        filename=os.path.basename(file),
        media_type="application/octet-stream",
    )


@router.post("/upload")
def upload_file(
    file: UploadFile = File(...),
    category: str = Query("일반행정"),
    classification: str = Query("일반"),
    description: str = Query(""),
    user: User = Depends(get_current_user_required),
    db: Session = Depends(get_db),
):
    """파일 업로드 (인증 필요)"""
    # 카테고리별 디렉토리 생성
    category_dir = os.path.join(settings.UPLOAD_DIR, category)
    os.makedirs(category_dir, exist_ok=True)

    # 파일 저장
    file_path = os.path.join(category_dir, file.filename)

    with open(file_path, "wb") as f:
        content = file.file.read()
        if len(content) > settings.MAX_UPLOAD_SIZE:
            raise HTTPException(status_code=413, detail="파일 크기 초과")
        f.write(content)

    # DB에 메타데이터 저장
    doc = Document(
        filename=file.filename,
        original_filename=file.filename,
        file_path=file_path,
        file_size=len(content),
        category=category,
        classification=classification,
        description=description,
        uploaded_by=user.username,
    )
    db.add(doc)

    # 감사 로그
    db.add(AuditLog(
        action="upload",
        username=user.username,
        target=file.filename,
        details=f"파일 업로드: {file.filename} ({category}/{classification})"
    ))
    db.commit()

    return {
        "status": "success",
        "file": {
            "id": doc.id,
            "filename": doc.original_filename,
            "size": doc.file_size,
            "category": doc.category,
        }
    }

    uploaded_by VARCHAR(50) REFERENCES users(username),

);

    id SERIAL PRIMARY KEY,
    action VARCHAR(50) NOT NULL,
    username VARCHAR(50),
    target VARCHAR(500),
    ip_address VARCHAR(45),
    timestamp TIMESTAMP DEFAULT NOW(),
    details TEXT
);

CREATE INDEX idx_documents_category ON documents(category);
CREATE INDEX idx_documents_uploaded_by ON documents(uploaded_by);

## 9. 설치 스크립트

```bash
#!/bin/bash
# I8 군 문서 저장 서버 설치 스크립트
set -euo pipefail

SCRIPT_DIR="$(cd "$(dirname "$0")" && pwd)"
APP_DIR="/opt/docstorage"

echo "========================================"
echo "  I8. 군 문서 저장 서버 설치"
echo "  docs.mnd.local / FastAPI + PostgreSQL"
echo "========================================"

# 1. 시스템 패키지
apt-get update && apt-get upgrade -y
apt-get install -y python3 python3-pip python3-venv nginx postgresql postgresql-client

# 2. PostgreSQL 설정
sudo -u postgres psql << 'EOSQL'
CREATE DATABASE mil_docstorage;
CREATE USER docstorage WITH PASSWORD 'DocStore2026!';
GRANT ALL PRIVILEGES ON DATABASE mil_docstorage TO docstorage;
\c mil_docstorage
GRANT ALL ON SCHEMA public TO docstorage;
EOSQL

# 3. 애플리케이션 배포
mkdir -p ${APP_DIR}
cp -r "${SCRIPT_DIR}/app/"* ${APP_DIR}/
cp "${SCRIPT_DIR}/.env.example" ${APP_DIR}/.env

# 4. Python 가상환경 및 의존성
cd ${APP_DIR}
python3 -m venv venv
source venv/bin/activate
pip install -r requirements.txt

# 5. DB 스키마 및 시드 데이터
sudo -u postgres psql -d mil_docstorage -f ${APP_DIR}/sql/schema.sql
sudo -u postgres psql -d mil_docstorage -f ${APP_DIR}/sql/seed.sql

# 6. 파일 저장소 및 샘플 파일 생성
mkdir -p ${APP_DIR}/files/{작전계획,군수보급,인사명령,통신보안,일반행정}

# 더미 파일 생성 (PDF/XLSX 대신 텍스트 파일로 시뮬레이션)
for category in 작전계획 군수보급 인사명령 통신보안 일반행정; do
    for i in $(seq 1 3); do
        dd if=/dev/urandom of="${APP_DIR}/files/${category}/sample_${i}.pdf" bs=1024 count=$((RANDOM % 4096 + 100)) 2>/dev/null
    done
done

# 실제 시드 데이터의 파일명과 매칭
cp "${SCRIPT_DIR}/seed_files/"* ${APP_DIR}/files/ 2>/dev/null || true

# 7. systemd 서비스
cat > /etc/systemd/system/docstorage.service << 'EOF'
[Unit]
Description=Military Document Storage Server
After=network.target postgresql.service

[Service]
Type=simple
User=www-data
Group=www-data
WorkingDirectory=/opt/docstorage
ExecStart=/opt/docstorage/venv/bin/uvicorn main:app --host 127.0.0.1 --port 8000
Restart=always
Environment=PATH=/opt/docstorage/venv/bin:/usr/bin

[Install]
WantedBy=multi-user.target
EOF

# 8. Nginx 설정
cat > /etc/nginx/sites-available/docstorage << 'EOF'
server {
    listen 80;
    server_name docs.mnd.local 192.168.92.228;

    client_max_body_size 50M;

    location /static/ {
        alias /opt/docstorage/static/;
    }

    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
EOF

ln -sf /etc/nginx/sites-available/docstorage /etc/nginx/sites-enabled/
rm -f /etc/nginx/sites-enabled/default

# 9. 권한 설정
chown -R www-data:www-data ${APP_DIR}

# 10. 서비스 시작
systemctl daemon-reload
systemctl enable --now docstorage nginx

echo "[완료] I8 군 문서 저장 서버 배포 완료"
echo "  - 웹 UI: http://192.168.92.228/"
echo "  - API 문서: http://192.168.92.228/docs"
echo "  - Path Traversal 취약: /api/download?file=../../etc/passwd"
```
